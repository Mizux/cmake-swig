endforeach()

  endforeach()

  foreach(FILE_NAME IN LISTS JAVA_SRCS)
    add_java_example(${FILE_NAME})
  endforeach()
endif()

if(BUILD_PYTHON)

  foreach(FILE_NAME IN LISTS PYTHON_SRCS)
    add_python_example(${FILE_NAME})
